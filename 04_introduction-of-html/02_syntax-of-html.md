<!-- +++++++++++++++ Syntax of HTML +++++++++++++++ -->
<!DOCTYPE html>
<html lang="en">
<head>
    <title>My Website</title>
</head>
<body>
    <h1>Hello World</h1>
</body>
</html>

<!-- +++++++++++++++ Quick Points +++++++++++++++ -->
=> <html> tag is the parent tag of <head> and <body> tag.
=> Most HTML elements have opening and closing tags.
=> Some tags are self-closing tags, like <br> and <hr>.

=> HTML is NOT case sensitive.
    <h1> = <H1>
	<p> = <P>
	<html> = <HTML>
	<head> = <HEAD>
	<body> = <BODY>
